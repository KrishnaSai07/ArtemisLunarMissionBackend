------------ Create Schema ---------

create schema artemisLunarMission;

----------- Create tables ----------

create table loginDetails(
	id int NOT NULL AUTO_INCREMENT,
	username varchar(255) NOT NULL,
    email varchar(255) NOT NULL,
    hashpassword varchar(255) NOT NULL,
    token varchar(255) null,
    shareLogIds varchar(255) null,
    PRIMARY KEY (id)

);

create table posts(
	id int NOT NULL AUTO_INCREMENT,
    subject varchar(255) NOT NULL,
    context varchar(21000) NULL,
    access varchar(255) NOT NULL,
    username varchar(255) NOT NULL,
    email varchar(255) NULL,
    files varchar(255) NULL,
    postTime timestamp,
    PRIMARY KEY (id)
);

---------------- LOAD THE DATA USING LOAD DATA INFILE WAY ------------------

Aftr cloning the project replace project path with 'Your Project Path' term below and run the below command on MySQL command line client

LOAD DATA LOCAL INFILE 'Your Project Path/credentials.txt' INTO TABLE artemisLunarMission.loginDetails
FIELDS TERMINATED BY ';'
LINES TERMINATED BY '\r\n';


LOAD DATA LOCAL INFILE 'Your Project Path/posts.txt' INTO TABLE artemisLunarMission.posts
FIELDS TERMINATED BY ';'
LINES TERMINATED BY '\r\n';
